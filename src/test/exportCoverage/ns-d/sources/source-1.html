336      *
337      * @param evtArray The events buffer that need to be managed and sorted via their identificators.

341     private void manageUpdates(ArrayList<PropertyChangeEvent> evtArray) throws LocalModelNotLoadedException {
342         for (PropertyChangeEvent evt : evtArray) {
343             switch (evt.getPropertyName()) {
344                 case "first-player":
345                     if (nickname.equals(evt.getNewValue()))
346                         setMyTurn(true);
347                     if (localModel != null) {
348                         try {
349                             localModel.updateModel(evt);
350                         } catch (BadFormattedLocalModelException badFormattedLocalModelException) {
351                             badFormattedLocalModelException.printStackTrace();
352                             badFormattedLocalModelException.printStackTrace();
353                         }
354                     } else {
355                         throw new LocalModelNotLoadedException();
356                     }
357                     break;
358                 case "init":
359                     localModel = (StrippedModel) evt.getNewValue();
360                     localModelLoaded = true;
361                     localModel.setUi(ui);
362                     try {
363                         view = StringNames.INGAME;
364                         ui.startGame();
365                     } catch (RemoteException e) {
366                         System.out.println(StringNames.REMOTE);
367                         userRegistered=false;
368                     }
369                     break;
370                 case "current-player":
371                     setInGame(true);
372                     if (nickname.equals(evt.getNewValue()))
373                         isMyTurn = true;
374                     if (localModel != null) {
375                         try {
376                             localModel.updateModel(evt);
377                         } catch (BadFormattedLocalModelException e) {
378                             System.out.println(StringNames.LOCAL_MODEL_ERROR);
379                         }
380                     } else {
381                         System.out.println(StringNames.LOCAL_MODEL_ERROR);
382                     }
383                     break;
384                 case "game-finished":
385                     try {
386                         leaveRoom();
387                     } catch (RemoteException e) {
388                         System.out.println(StringNames.REMOTE);
389                     } catch (UserNotInRoomException e) {
390                         System.out.println(StringNames.USER_NOT_IN_ROOM);
391                     } catch (UserNotRegisteredException e) {
392                         System.out.println(StringNames.USER_NOT_REGISTERED);
393                     }
394                     ui.gameOver((String) evt.getOldValue(), (String) evt.getNewValue());
395                     break;
396                 default:
397                     if (localModel != null) {
398                         try {
399                             localModel.updateModel(evt);
400                         } catch (BadFormattedLocalModelException e) {
401                             System.out.println(StringNames.LOCAL_MODEL_ERROR);
402                         }
403                     } else {
404                         System.out.println(StringNames.LOCAL_MODEL_ERROR);
405                     }
