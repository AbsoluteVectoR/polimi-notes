389      * find the players who ping the server. If the client has correctly marked its ping value on 'up'
390      * the server set it down (pong). If the client has disconnected, the server found the ping value has 'down'
391      * and de-register the client.
392      */
393     private synchronized void findDisconnectedUsers() {
394         ArrayList<ClientConnection> usersToRemove = new ArrayList<>();
395 
396         synchronized (users) {
397             for (ClientConnection client : users.values()) {
398                 if (!client.isUp()) {
399                     usersToRemove.add(client);
400                 } else client.setDown();
401             }
402             for (ClientConnection clientToRemove : usersToRemove) {
403                 try {
404                     deregisterConnection(clientToRemove.getNickname());
405                     //inside the server can't be 'remoteException' and UserNotRegistered is not harmful in this case

407                 }
408             }
409         }
